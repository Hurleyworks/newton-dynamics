219 template<class T>
220 void ndArray<T>::Resize(ndInt32 newSize)
221 {
222  // note: I know some tolls will detect a warning here
223  // because it is copy object from one array
224  // to another with out calling the copy constructor
225  // and destructor, but the ndArray is designed for
226  // high performance memory resizing of struct,
227  // if an application needs to use an array with
228  // for general purpose classes,
229  // please use standart lib std::vector
230  if (newSize > m_capacity || (m_capacity == 0))
231  {
232  newSize = ndMax(newSize, 16);
233  T* const newArray = (T*)ndMemory::Malloc(ndInt32(sizeof(T) * newSize));
234  if (m_array)
235  {
236  CopyData(newArray, m_array, m_size);
237  ndMemory::Free(m_array);
238  }
239  m_array = newArray;
240  m_capacity = newSize;
241  }
242  else if (newSize < m_capacity)
243  {
244  newSize = ndMax(newSize, 16);
245  T* const newArray = (T*)ndMemory::Malloc(ndInt32(sizeof(T) * newSize));
246  if (m_array)
247  {
248  CopyData(newArray, m_array, newSize);
249  ndMemory::Free(m_array);
250  }
251  m_size = newSize;
252  m_array = newArray;
253  m_capacity = newSize;
254  }
255 }
