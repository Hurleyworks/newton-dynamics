31 // the solver is a RK order 4, but instead of weighting the intermediate derivative by the usual 1/6, 1/3, 1/3, 1/6 coefficients
32 // I am using 1/4, 1/4, 1/4, 1/4.
33 // This is correct.The weighting coefficients of any RK method comes from an arbitrary criteria
34 // solving for a set of linear equation on the coefficients.
35 // The standard coefficients just happen to lead to an accurate result because they are optimal
36 // with respect to a second order solution.For differential equations of higher order, then is not
37 // clear if the traditional weigh factors are any better than any other set of weighting factors.
38 // A different set of coefficient generates results that are not much different than the optimal set,
39 // but it allows for simpler calculation of the intermediate derivatives and also for less intermediate memory.
40 // For more detail on the derivation of the Runge Kutta coefficients you can go to:
41 // http://pathfinder.scar.utoronto.ca/~dyer/csca57/book_P/node51.html

205 inline void ndDynamicsUpdate::ClearJacobianBuffer(ndInt32 count, ndJacobian* const buffer) const
206 {
207  const ndVector zero(ndVector::m_zero);
208  ndVector* const dst = &buffer[0].m_linear;
209  for (ndInt32 i = 0; i < count; ++i)
210  {
211  dst[i * 2 + 0] = zero;
212  dst[i * 2 + 1] = zero;
213  }
214 }

217 {

220  ClearJacobianBuffer(sizeInJacobian, (ndJacobian*)buffer);

228 inline ndBodyKinematic* ndDynamicsUpdate::FindRootAndSplit(ndBodyKinematic* const body)
229 {
230  ndBodyKinematic* node = body;
231  while (node->m_islandParent != node)
232  {
233  ndBodyKinematic* const prev = node;
234  node = node->m_islandParent;
235  prev->m_islandParent = node->m_islandParent;
236  }
237  return node;
238 }
