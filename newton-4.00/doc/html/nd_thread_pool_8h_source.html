182 template <typename Function>
183 void ndThreadPool::ParallelExecute(const Function& ndFunction)
184 {
185  const ndInt32 threadCount = GetThreadCount();
186  ndTaskImplement<Function>* const jobsArray = ndAlloca(ndTaskImplement<Function>, threadCount);
187 
188  for (ndInt32 i = 0; i < threadCount; ++i)
189  {
190  ndTaskImplement<Function>* const job = &jobsArray[i];
191  new (job) ndTaskImplement<Function>(i, this, ndFunction);
192  }
193 
194  if (m_count > 0)
195  {
196  #ifdef D_USE_THREAD_EMULATION
197  for (ndInt32 i = 0; i < threadCount; ++i)
198  {
199  ndTaskImplement<Function>* const job = &jobsArray[i];
200  ndFunction(job->m_threadIndex, job->m_threadCount);
201  }
202  #else
203  for (ndInt32 i = 0; i < m_count; ++i)
204  {
205  ndTaskImplement<Function>* const job = &jobsArray[i];
206  m_workers[i].m_task.store(job);
207  }
208 
209  ndTaskImplement<Function>* const job = &jobsArray[m_count];
210  ndFunction(job->m_threadIndex, job->m_threadCount);
211 
212  bool jobsInProgress = true;
213  do
214  {
215  bool inProgess = false;
216  for (ndInt32 i = 0; i < m_count; ++i)
217  {
218  inProgess = inProgess | (m_workers[i].m_task.load() != nullptr);
219  }
220  jobsInProgress = jobsInProgress & inProgess;
221  if (jobsInProgress)
222  {
223  ndYield();
224  }
225  } while (jobsInProgress);
226  #endif
227  }
228  else
229  {
230  ndTaskImplement<Function>* const job = &jobsArray[0];
231  ndFunction(job->m_threadIndex, job->m_threadCount);
232  }
233 }
234 
235 #endif
